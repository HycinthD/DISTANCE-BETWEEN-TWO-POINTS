import math
def dis(x1,x2,y1, y2):
    d=((x2-x1)**2+(y2-y1)**2)
    g=math.sqrt(d)
    return g

x1=4 
x2=10
y1=2
y2=6
g=dis(x1,x2,y1,y2) 
print("{:.2f}".format(g))
